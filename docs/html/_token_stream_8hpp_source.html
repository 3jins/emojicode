92  Token advanceLexer() {
93  skippedBlankLine_ = false;
94  auto temp = std::move(nextToken_);
95  while (true) {
96  if (lexer_.continues()) {
97  nextToken_ = lexer_.lex();
98  if (nextToken_.type() == TokenType::BlankLine) {
99  skippedBlankLine_ = true;
100  continue;
101  }
102  else if (nextToken_.type() == TokenType::SinglelineComment ||
103  nextToken_.type() == TokenType::MultilineComment) {
104  nextToken_.position().file->addComment(std::move(nextToken_));
105  continue;
106  }
107  else if (nextToken_.type() == TokenType::LineBreak) {
108  continue;
109  }
110  }
111  else {
112  moreTokens_ = false;
113  }
114  break;
115  }

117  }
