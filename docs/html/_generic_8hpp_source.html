80  void checkGenericArguments(const TypeContext &typeContext, const std::vector<Type> &args, const SourcePosition &p) {
81  if (args.size() - offset_ != genericParameters().size()) {
82  throw CompilerError(p, "Expected ", genericParameters().size(), " generic arguments, but ",
83  args.size(), " are provided.");
84  }

86  for (size_t i = offset_; i < args.size(); i++) {

88  throw CompilerError(p, "Generic argument ", i + 1, " of type ",
89  args[i].toString(typeContext), " is not compatible to constraint ",
