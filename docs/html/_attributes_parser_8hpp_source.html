30 template <Attribute ...Attributes>
31 class AttributeParser {
32 public:
33  AttributeParser& allow(Attribute attr) { found_.find(attr)->second.allowed = true; return *this; }
34  bool has(Attribute attr) const { return found_.find(attr)->second.found; }
35 
36  void check(const SourcePosition &p, Compiler *compiler) const {
37  for (auto &pair : found_) {
38  if (!pair.second.allowed && pair.second.found) {
39  auto name = utf8(std::u32string(1, static_cast<char32_t>(pair.first)));
40  compiler->error(CompilerError(p, "Attribute ", name, " not applicable."));
41  }
42  }
43  }

45  AttributeParser& parse(TokenStream *stream) {
46  for (auto attr : order_) {
47  auto found = test(stream, attr);
48  found_.emplace(attr, FoundAttribute(found));
49  }
50  return *this;
51  }
52 private:
53  struct FoundAttribute {
54  explicit FoundAttribute(bool found) : found(found) {}
55  bool found;
56  bool allowed = false;
57  };
58 
59  bool test(TokenStream *stream, Attribute attr) const {
60  switch (attr) {
61  case Attribute::Mutating:
62  return stream->consumeTokenIf(TokenType::Mutable);
63  case Attribute ::Unsafe:
64  return stream->consumeTokenIf(TokenType::Unsafe);
65  case Attribute::StaticOnType:
66  return stream->consumeTokenIf(TokenType::Class);
67  default:
68  return stream->consumeTokenIf(static_cast<char32_t>(attr));
69  }
70  }
71 
72  constexpr static const std::array<Attribute, sizeof...(Attributes)> order_ = { Attributes... };
73  std::map<Attribute, FoundAttribute> found_;
74 };
